53enum class texture_access
54{
55 non_lockable = SDL_TEXTUREACCESS_STATIC,
56 streaming = SDL_TEXTUREACCESS_STREAMING,
57 target = SDL_TEXTUREACCESS_TARGET
58};
59
60[[nodiscard]] constexpr auto to_string(const texture_access access) -> std::string_view
61{
62 switch (access) {
63 case texture_access::non_lockable:
64 return "non_lockable";
65
66 case texture_access::streaming:
67 return "streaming";
68
69 case texture_access::target:
70 return "target";
71
72 default:
73 throw exception{"Did not recognize texture access!"};
74 }
75}
76
77inline auto operator<<(std::ostream& stream, const texture_access access) -> std::ostream&
78{
79 return stream << to_string(access);
80}

127template <typename T>
128class basic_texture final
129{
130 public:
138 explicit basic_texture(maybe_owner<SDL_Texture*> source) noexcept(detail::is_handle<T>)
139 : mTexture{source}
140 {
141 if constexpr (detail::is_owner<T>) {
142 if (!mTexture) {
143 throw exception{"Cannot create texture from null pointer!"};
144 }
145 }
146 }

291template <typename T>
292[[nodiscard]] auto to_string(const basic_texture<T>& texture) -> std::string
293{
294#if CENTURION_HAS_FEATURE_FORMAT
295 return std::format("texture(data: {}, width: {}, height: {})",
296 detail::address_of(texture.get()),
297 texture.width(),
298 texture.height());
299#else
300 return "texture(data: " + detail::address_of(texture.get()) +
301 ", width: " + std::to_string(texture.width()) +
302 ", height: " + std::to_string(texture.height()) + ")";
303#endif // CENTURION_HAS_FEATURE_FORMAT
304}
