87[[nodiscard]] inline auto to_string(const platform_id id) -> std::string_view
88{

170[[nodiscard]] inline auto platform_name() -> std::optional<std::string>
171{
172 std::string name{SDL_GetPlatform()};
173 if (name != "Unknown") {
174 return name;
175 }
176 else {
177 return std::nullopt;
178 }
179}

181[[nodiscard]] inline auto current_platform() noexcept -> platform_id
182{
183 const auto name = platform_name();
184 if (name == "Windows") {
185 return platform_id::windows;
186 }
187 else if (name == "Mac OS X") {
188 return platform_id::macos;
189 }
190 else if (name == "Linux") {
191 return platform_id::linux_os;
192 }
193 else if (name == "iOS") {
194 return platform_id::ios;
195 }
196 else if (name == "Android") {
197 return platform_id::android;
198 }
199 else {
200 return platform_id::unknown;
201 }
202}
