189class simd_block final
190{
191 public:
192 using size_type = std::size_t;
193
201 explicit simd_block(const size_type size) noexcept : mData{SDL_SIMDAlloc(size)} {}
202
203#if SDL_VERSION_ATLEAST(2, 0, 14)
204
212 void reallocate(const size_type size) noexcept
213 {
214 /* We temporarily release the ownership of the pointer in order to avoid a double
215 delete, since the reallocation will free the previously allocated memory. */
216 auto* ptr = mData.release();
217 mData.reset(SDL_SIMDRealloc(ptr, size));
218 }
219
220#endif // SDL_VERSION_ATLEAST(2, 0, 14)
221
222 [[nodiscard]] auto data() noexcept -> void* { return mData.get(); }
223
224 [[nodiscard]] auto data() const noexcept -> const void* { return mData.get(); }
225
231 explicit operator bool() const noexcept { return mData != nullptr; }
232
233 private:
234 struct simd_deleter final
235 {
236 void operator()(void* ptr) noexcept { SDL_SIMDFree(ptr); }
237 };
238 std::unique_ptr<void, simd_deleter> mData;
239};
