97[[nodiscard]] inline auto is_active(const key_mod modifiers, const uint16 currentMask) noexcept
98 -> bool
99{
100 if (modifiers == key_mod::none) {
101 return !currentMask;
102 }
103 else {
104 return currentMask & to_underlying(modifiers);
105 }
106}
107
108[[nodiscard]] inline auto is_only_active(const key_mod modifiers,
109 const uint16 currentMask) noexcept -> bool
110{
111 if (modifiers == key_mod::none) {
112 return !currentMask;
113 }
114
115 const auto mask = to_underlying(modifiers);
116 const auto hits = currentMask & mask;
117
118 if (hits != mask) {
119 return false; // The specified modifiers were a combo that wasn't fully active
120 }
121 else {
122 const auto others = currentMask & ~hits;
123 return hits && !others;
124 }
125}
126
127[[nodiscard]] inline auto is_only_subset_active(const key_mod modifiers,
128 const uint16 currentMask) noexcept -> bool
129{
130 if (modifiers == key_mod::none) {
131 return !currentMask;
132 }
133
134 const auto mask = to_underlying(modifiers);
135
136 const auto hits = currentMask & mask;
137 const auto others = currentMask & ~hits;
138
139 return hits && !others;
140}

142} // namespace detail

157}

162}

164[[nodiscard]] inline auto to_string(const key_mod mods) -> std::string
165{
166 if (mods == key_mod::none) {
167 return "none";
168 }
169
170 const auto mask = to_underlying(mods);
171 std::stringstream stream;
172
173 auto check = [&stream, mask, count = 0](const key_mod mod, const char* name) mutable {
174 if (mask & to_underlying(mod)) {
175 if (count != 0) {
176 stream << ',';
177 }
178
179 stream << name;
180 ++count;
181 }
182 };
183
184 check(key_mod::lshift, "lshift");
185 check(key_mod::rshift, "rshift");
186
187 check(key_mod::lctrl, "lctrl");
188 check(key_mod::rctrl, "rctrl");
189
190 check(key_mod::lalt, "lalt");
191 check(key_mod::ralt, "ralt");
192
193 check(key_mod::lgui, "lgui");
194 check(key_mod::rgui, "rgui");
195
196 check(key_mod::num, "num");
197 check(key_mod::caps, "caps");
198 check(key_mod::mode, "mode");
199
200 return stream.str();
201}

580class keyboard final
581{
582 public:
583 keyboard() noexcept { mState = SDL_GetKeyboardState(&mKeyCount); }
584
588 void refresh() { std::copy(mState, mState + mKeyCount, mPrevious.begin()); }
589
599 [[nodiscard]] auto is_pressed(const scan_code& code) const noexcept -> bool
600 {
601 return check(code, [this](const SDL_Scancode sc) noexcept { return mState[sc]; });
602 }
603
605 [[nodiscard]] auto is_pressed(const key_code& code) const noexcept -> bool
606 {
607 return is_pressed(code.to_scancode());
608 }
609
619 [[nodiscard]] auto is_held(const scan_code& code) const noexcept(on_msvc) -> bool
620 {
621 return check(code, [this](const SDL_Scancode sc) noexcept(on_msvc) {
622 return mState[sc] && mPrevious[sc];
623 });
624 }
625
627 [[nodiscard]] auto is_held(const key_code& code) const noexcept(on_msvc) -> bool
628 {
629 return is_held(code.to_scancode());
630 }
631
641 [[nodiscard]] auto just_pressed(const scan_code& code) const noexcept(on_msvc) -> bool
642 {
643 return check(code, [this](const SDL_Scancode sc) noexcept(on_msvc) {
644 return mState[sc] && !mPrevious[sc];
645 });
646 }
647
649 [[nodiscard]] auto just_pressed(const key_code& code) const noexcept(on_msvc) -> bool
650 {
651 return just_pressed(code.to_scancode());
652 }
653
663 [[nodiscard]] auto just_released(const scan_code& code) const noexcept(on_msvc) -> bool
664 {
665 return check(code, [this](const SDL_Scancode sc) noexcept(on_msvc) {
666 return !mState[sc] && mPrevious[sc];
667 });
668 }
669
671 [[nodiscard]] auto just_released(const key_code& code) const noexcept(on_msvc) -> bool
672 {
673 return just_released(code.to_scancode());
674 }
675
681 [[nodiscard]] auto size() const noexcept -> int { return mKeyCount; }
682
683 private:
684 const uint8* mState{};
685 std::array<uint8, cen::scan_code::count()> mPrevious{};
686 int mKeyCount{};
687
688 template <typename Predicate>
689 auto check(const cen::scan_code& code, Predicate&& predicate) const
690 noexcept(noexcept(predicate(code.get()))) -> bool
691 {
692 const auto sc = code.get();
693 if (sc >= 0 && sc < mKeyCount) {
694 return predicate(sc);
695 }
696 else {
697 return false;
698 }
699 }
700};
