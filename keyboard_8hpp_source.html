76[[nodiscard]] inline auto is_active(const key_mod modifiers, const uint16 currentMask) noexcept
77 -> bool
78{
79 if (modifiers == key_mod::none) {
80 return !currentMask;
81 }
82 else {
83 return currentMask & to_underlying(modifiers);
84 }
85}
86
87[[nodiscard]] inline auto is_only_active(const key_mod modifiers,
88 const uint16 currentMask) noexcept -> bool
89{
90 if (modifiers == key_mod::none) {
91 return !currentMask;
92 }
93
94 const auto mask = to_underlying(modifiers);
95 const auto hits = currentMask & mask;
96
97 if (hits != mask) {
98 return false; // The specified modifiers were a combo that wasn't fully active
99 }
100 else {
101 const auto others = currentMask & ~hits;
102 return hits && !others;
103 }
104}
105
106[[nodiscard]] inline auto is_only_subset_active(const key_mod modifiers,
107 const uint16 currentMask) noexcept -> bool
108{
109 if (modifiers == key_mod::none) {
110 return !currentMask;
111 }
112
113 const auto mask = to_underlying(modifiers);
114
115 const auto hits = currentMask & mask;
116 const auto others = currentMask & ~hits;
117
118 return hits && !others;
119}
120
121} // namespace detail

128[[nodiscard]] constexpr auto operator&(const key_mod a, const key_mod b) noexcept -> key_mod

130 return static_cast<key_mod>(to_underlying(a) & to_underlying(b));
131}

133[[nodiscard]] constexpr auto operator|(const key_mod a, const key_mod b) noexcept -> key_mod
134{
135 return static_cast<key_mod>(to_underlying(a) | to_underlying(b));
136}

138[[nodiscard]] inline auto to_string(const key_mod mods) -> std::string
139{
140 if (mods == key_mod::none) {
141 return "none";
142 }
143
144 const auto mask = to_underlying(mods);
145 std::stringstream stream;
146
147 auto check = [&stream, mask, count = 0](const key_mod mod, const char* name) mutable {
148 if (mask & to_underlying(mod)) {
149 if (count != 0) {
150 stream << ',';
151 }
152
153 stream << name;
154 ++count;
155 }
156 };
157
158 check(key_mod::lshift, "lshift");
159 check(key_mod::rshift, "rshift");
160
161 check(key_mod::lctrl, "lctrl");
162 check(key_mod::rctrl, "rctrl");
163
164 check(key_mod::lalt, "lalt");
165 check(key_mod::ralt, "ralt");
166
167 check(key_mod::lgui, "lgui");
168 check(key_mod::rgui, "rgui");
169
170 check(key_mod::num, "num");
171 check(key_mod::caps, "caps");
172 check(key_mod::mode, "mode");
173
174 return stream.str();
175}

177inline auto operator<<(std::ostream& stream, const key_mod& mods) -> std::ostream&
178{
179 return stream << to_string(mods);
180}

446class keyboard final
447{
448 public:
449 keyboard() noexcept { mState = SDL_GetKeyboardState(&mKeyCount); }
450
451 void refresh() { std::copy(mState, mState + mKeyCount, mPrevious.begin()); }
452
454 [[nodiscard]] auto is_pressed(const scan_code& code) const noexcept -> bool
455 {
456 return check(code, [this](const SDL_Scancode sc) noexcept { return mState[sc]; });
457 }
458
459 [[nodiscard]] auto is_pressed(const key_code& code) const noexcept -> bool
460 {
461 return is_pressed(code.to_scancode());
462 }
463
465 [[nodiscard]] auto is_held(const scan_code& code) const noexcept(on_msvc) -> bool
466 {
467 return check(code, [this](const SDL_Scancode sc) noexcept(on_msvc) {
468 return mState[sc] && mPrevious[sc];
469 });
470 }
471
472 [[nodiscard]] auto is_held(const key_code& code) const noexcept(on_msvc) -> bool
473 {
474 return is_held(code.to_scancode());
475 }
476
478 [[nodiscard]] auto just_pressed(const scan_code& code) const noexcept(on_msvc) -> bool
479 {
480 return check(code, [this](const SDL_Scancode sc) noexcept(on_msvc) {
481 return mState[sc] && !mPrevious[sc];
482 });
483 }
484
485 [[nodiscard]] auto just_pressed(const key_code& code) const noexcept(on_msvc) -> bool
486 {
487 return just_pressed(code.to_scancode());
488 }
489
491 [[nodiscard]] auto just_released(const scan_code& code) const noexcept(on_msvc) -> bool
492 {
493 return check(code, [this](const SDL_Scancode sc) noexcept(on_msvc) {
494 return !mState[sc] && mPrevious[sc];
495 });
496 }
497
498 [[nodiscard]] auto just_released(const key_code& code) const noexcept(on_msvc) -> bool
499 {
500 return just_released(code.to_scancode());
501 }
502
503 [[nodiscard]] auto size() const noexcept -> int { return mKeyCount; }
504
505 private:
506 const uint8* mState{};
507 std::array<uint8, cen::scan_code::count()> mPrevious{};
508 int mKeyCount{};
509
510 template <typename Predicate>
511 auto check(const cen::scan_code& code, Predicate&& predicate) const
512 noexcept(noexcept(predicate(code.get()))) -> bool
513 {
514 const auto sc = code.get();
515 if (sc >= 0 && sc < mKeyCount) {
516 return predicate(sc);
517 }
518 else {
519 return false;
520 }
521 }
522};

527}

529[[nodiscard]] inline auto to_string(const keyboard& keyboard) -> std::string
530{
531#if CENTURION_HAS_FEATURE_FORMAT
532 return std::format("keyboard(size: {})", keyboard.size());
533#else
534 return "keyboard(size: " + std::to_string(keyboard.size()) + ")";
535#endif // CENTURION_HAS_FEATURE_FORMAT
536}
537
538inline auto operator<<(std::ostream& stream, const keyboard& keyboard) -> std::ostream&
539{
540 return stream << to_string(keyboard);
541}
