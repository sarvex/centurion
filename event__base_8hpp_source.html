128[[nodiscard]] constexpr auto is_user_event(const event_type type) noexcept -> bool
129{
130 const auto raw = to_underlying(type);
131 return raw >= SDL_USEREVENT && raw < SDL_LASTEVENT;
132}

342template <typename T>
343class event_base
344{
345 public:
346 explicit event_base(const event_type type)
347 {
348 set_timestamp(u32ms{SDL_GetTicks()});
349 set_type(type);
350 }
351
352 explicit event_base(const T& event) noexcept : mEvent{event} {}
353
354 explicit event_base(T&& event) noexcept : mEvent{std::move(event)} {}
355
356 void set_timestamp(const u32ms timestamp) noexcept(on_msvc)
357 {
358 mEvent.timestamp = timestamp.count();
359 }
360
361 void set_type(const event_type type) noexcept { mEvent.type = to_underlying(type); }
362
363 [[nodiscard]] auto timestamp() const -> u32ms { return u32ms{mEvent.timestamp}; }
364
365 [[nodiscard]] auto type() const noexcept -> event_type { return event_type{mEvent.type}; }
366
367 [[nodiscard]] auto get() const noexcept -> const T& { return mEvent; }
368
369 protected:
370 T mEvent{};
371};

375[[nodiscard]] auto as_sdl_event(const event_base<T>& event) -> SDL_Event;
