666template <typename T>
667[[nodiscard]] auto to_string(const basic_joystick<T>& joystick) -> std::string
668{
669 const char* serial{};
670 if constexpr (detail::sdl_version_at_least(2, 0, 14)) {
671 serial = joystick.serial();
672 }
673
674#if CENTURION_HAS_FEATURE_FORMAT
675 return std::format("joystick(data: {}, id: {}, name: '{}', serial: {})",
676 detail::address_of(joystick.get()),
677 joystick.GetID(),
678 str_or_na(joystick.GetName()),
679 str_or_na(serial));
680#else
681 return "joystick(data: " + detail::address_of(joystick.get()) +
682 ", id: " + std::to_string(joystick.id()) + ", name: '" + str_or_na(joystick.name()) +
683 ", serial: " + str_or_na(serial) + "')";
684#endif // CENTURION_HAS_FEATURE_FORMAT
685}
