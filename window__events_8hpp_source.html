136class window_event final : public event_base<SDL_WindowEvent>
137{
138 public:
139 window_event() : event_base{event_type::window} {}
140
141 explicit window_event(const SDL_WindowEvent& event) noexcept : event_base{event} {}
142
143 void set_event_id(const window_event_id id) noexcept
144 {
145 mEvent.event = static_cast<uint8>(id);
146 }
147
148 void set_data1(const int32 value) noexcept { mEvent.data1 = value; }
149
150 void set_data2(const int32 value) noexcept { mEvent.data2 = value; }
151
152 [[nodiscard]] auto event_id() const noexcept -> window_event_id
153 {
154 return static_cast<window_event_id>(mEvent.event);
155 }
156
157 [[nodiscard]] auto data1() const noexcept -> int32 { return mEvent.data1; }
158
159 [[nodiscard]] auto data2() const noexcept -> int32 { return mEvent.data2; }
160};
