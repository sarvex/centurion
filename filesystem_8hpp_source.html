493[[nodiscard]] inline auto preferred_path(const char* org, const char* app) -> sdl_string
494{
495 /* Looking at the SDL source code, it actually seems fine to supply a null
496 string for the organization name. However, I haven't been able to find any
497 documentation providing this guarantee, so we simply disallow null
498 organization names. */
499 assert(org);
500 assert(app);
501 return sdl_string{SDL_GetPrefPath(org, app)};
502}
