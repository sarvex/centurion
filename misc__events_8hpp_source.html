65enum class display_event_id
66{
67 none = SDL_DISPLAYEVENT_NONE,
68 orientation = SDL_DISPLAYEVENT_ORIENTATION,
69 connected = SDL_DISPLAYEVENT_CONNECTED,
70 disconnected = SDL_DISPLAYEVENT_DISCONNECTED,
71};
72
75
76[[nodiscard]] constexpr auto to_string(const display_event_id id) -> std::string_view
77{
78 switch (id) {
79 case display_event_id::none:
80 return "none";
81
82 case display_event_id::orientation:
83 return "orientation";
84
85 case display_event_id::connected:
86 return "connected";
87
88 case display_event_id::disconnected:
89 return "disconnected";
90
91 default:
92 throw exception{"Did not recognize display event ID!"};
93 }
94}
95
96inline auto operator<<(std::ostream& stream, const display_event_id id) -> std::ostream&
97{
98 return stream << to_string(id);
99}
