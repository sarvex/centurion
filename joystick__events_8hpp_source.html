42enum class joy_hat_position : uint8
43{
44 left_up = SDL_HAT_LEFTUP,
45 left = SDL_HAT_LEFT,
46 left_down = SDL_HAT_LEFTDOWN,
47 up = SDL_HAT_UP,
48 centered = SDL_HAT_CENTERED,
49 down = SDL_HAT_DOWN,
50 right_up = SDL_HAT_RIGHTUP,
51 right = SDL_HAT_RIGHT,
52 right_down = SDL_HAT_RIGHTDOWN
53};

58[[nodiscard]] constexpr auto to_string(const joy_hat_position position) -> std::string_view
59{
60 switch (position) {
61 case joy_hat_position::left_up:
62 return "left_up";
63
64 case joy_hat_position::left:
65 return "left";
66
67 case joy_hat_position::left_down:
68 return "left_down";
69
70 case joy_hat_position::up:
71 return "up";
72
73 case joy_hat_position::centered:
74 return "centered";
75
76 case joy_hat_position::down:
77 return "down";
78
79 case joy_hat_position::right_up:
80 return "right_up";
81
82 case joy_hat_position::right:
83 return "right";
84
85 case joy_hat_position::right_down:
86 return "right_down";
87
88 default:
89 throw exception{"Did not recognize joystick hat position!"};
90 }
91}
