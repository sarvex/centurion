63enum class renderer_flip
64{
65 none = SDL_FLIP_NONE,
66 horizontal = SDL_FLIP_HORIZONTAL,
67 vertical = SDL_FLIP_VERTICAL

70[[nodiscard]] constexpr auto to_string(const renderer_flip flip) -> std::string_view
71{
72 switch (flip) {
73 case renderer_flip::none:
74 return "none";
75
76 case renderer_flip::horizontal:
77 return "horizontal";
78
79 case renderer_flip::vertical:
80 return "vertical";
81
82 default:
83 throw exception{"Did not recognize renderer flip value!"};
84 }
85}

250 {

252 using value_t = typename point_t::value_type; // either int or float

259 return SDL_RenderDrawLines(get(), first, isize(container)) == 0;

265 else {
266 return failure;

281 template <typename X>
282 void draw_circle(const basic_point<X>& position, const float radius) noexcept
283 {
284 using value_t = typename basic_point<X>::value_type;
285
286 auto error = -radius;
287 auto x = radius - 0.5f;
288 auto y = 0.5f;
289
290 const auto cx = static_cast<float>(position.x()) - 0.5f;
291 const auto cy = static_cast<float>(position.y()) - 0.5f;
292
293 while (x >= y) {
294 draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy + y)});
295 draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy + x)});
296
297 if (x != 0) {
298 draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy + y)});
299 draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy - x)});
300 }
301
302 if (y != 0) {
303 draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy - y)});
304 draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy + x)});
305 }
306
307 if (x != 0 && y != 0) {
308 draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy - y)});
309 draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy - x)});
310 }
311
312 error += y;
313 ++y;
314 error += y;
315
316 if (error >= 0) {
317 --x;
318 error -= x;
319 error -= x;
320 }
321 }
322 }

324 template <typename X>
325 void fill_circle(const basic_point<X>& center, const float radius)
326 {
327 const auto cx = static_cast<float>(center.x());
328 const auto cy = static_cast<float>(center.y());
329
330 for (auto dy = 1.0f; dy <= radius; dy += 1.0f) {
331 const auto dx = std::floor(std::sqrt((2.0f * radius * dy) - (dy * dy)));
332 draw_line<float>({cx - dx, cy + dy - radius}, {cx + dx, cy + dy - radius});
333 draw_line<float>({cx - dx, cy - dy + radius}, {cx + dx, cy - dy + radius});
334 }
335 }

337 template <typename X, typename Y>
338 auto render(const basic_texture<X>& texture, const basic_point<Y>& pos) noexcept -> result
339 {
340 if constexpr (basic_point<Y>::floating) {
341 const auto size = texture.size().as_f();
342 const SDL_FRect dst{pos.x(), pos.y(), size.width, size.height};
343 return SDL_RenderCopyF(get(), texture.get(), nullptr, &dst) == 0;
344 }
345 else {
346 const SDL_Rect dst{pos.x(), pos.y(), texture.width(), texture.height()};
347 return SDL_RenderCopy(get(), texture.get(), nullptr, &dst) == 0;
348 }

401 template <typename X, typename Y, typename Z>
402 auto render(const basic_texture<X>& texture,
403 const irect& src,
404 const basic_rect<Y>& dst,
405 const double angle,
406 const basic_point<Z>& center,
407 const renderer_flip flip) noexcept -> result
408 {
409 static_assert(std::is_same_v<typename basic_rect<Y>::value_type,
410 typename basic_point<Z>::value_type>,
411 "Destination rectangle and center point must have the same "
412 "value types (int or float)!");
413
414 if constexpr (basic_rect<Y>::floating) {
415 return SDL_RenderCopyExF(get(),
416 texture.get(),
417 src.data(),
418 dst.data(),
419 angle,
420 center.data(),
421 static_cast<SDL_RendererFlip>(flip)) == 0;
422 }
423 else {
424 return SDL_RenderCopyEx(get(),
425 texture.get(),
426 src.data(),
427 dst.data(),
428 angle,
429 center.data(),
430 static_cast<SDL_RendererFlip>(flip)) == 0;
431 }
432 }

647 [[nodiscard]] auto capture(const pixel_format format) const -> surface
648 {
649 surface image{output_size(), format};
650
651 if (!image.lock()) {
652 throw sdl_error{};
653 }
654
655 if (const auto res =
656 SDL_RenderReadPixels(get(), nullptr, 0, image.pixel_data(), image.pitch());
657 res == -1) {
658 throw sdl_error{};
659 }
660
661 image.unlock();
662 return image;
663 }

707class renderer_info final

716 [[nodiscard]] auto supported_flags() const noexcept -> uint32 { return mInfo.flags; }

718 [[nodiscard]] auto has_vsync() const noexcept -> bool
719 {
720 return supported_flags() & SDL_RENDERER_PRESENTVSYNC;
721 }
722
723 [[nodiscard]] auto has_target_textures() const noexcept -> bool
724 {
725 return supported_flags() & SDL_RENDERER_TARGETTEXTURE;

729 {
730 return supported_flags() & SDL_RENDERER_ACCELERATED;
731 }
732
733 [[nodiscard]] auto is_software() const noexcept -> bool
734 {
735 return supported_flags() & SDL_RENDERER_SOFTWARE;
736 }
737
738 [[nodiscard]] auto format_count() const noexcept -> size_type
739 {
740 return static_cast<size_type>(mInfo.num_texture_formats);
741 }

743 [[nodiscard]] auto get_format(const size_type index) const -> pixel_format
744 {
745 if (index < format_count()) {
746 return static_cast<pixel_format>(mInfo.texture_formats[index]);
747 }
748 else {
749 throw exception{"Invalid pixel format index!"};
750 }
751 }
752

761 }
762
763 [[nodiscard]] auto max_texture_height() const noexcept -> int
764 {
765 return mInfo.max_texture_height;
766 }
767
768 [[nodiscard]] auto name() const noexcept -> const char* { return mInfo.name; }

770 private:

773 explicit renderer_info(const SDL_RendererInfo info) noexcept : mInfo{info} {}

791template <typename T>
792[[nodiscard]] auto get_info(const basic_renderer<T>& renderer) noexcept
793 -> std::optional<renderer_info>
794{
795 SDL_RendererInfo info{};
796 if (SDL_GetRendererInfo(renderer.get(), &info) == 0) {
797 return renderer_info{info};
798 }
